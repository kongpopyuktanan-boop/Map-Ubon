    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-app.js";
        import { getFirestore, collection, addDoc } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js";

        const firebaseConfig = {
            apiKey: "AIzaSyCMfq2e1gmsz3h3xbMV5o26GY_P0JXAgdU",
            authDomain: "mapppp-2258d.firebaseapp.com",
            projectId: "mapppp-2258d",
            storageBucket: "mapppp-2258d.appspot.com",
            messagingSenderId: "785706585070",
            appId: "1:785706585070:web:af4cd7c96530be9a05dcff",
            measurementId: "G-X8TFN4065K"
        };

        const app = initializeApp(firebaseConfig);
        const db = getFirestore(app);

        const submitBtn = document.getElementById('submitBtn');
        const alertMessage = document.getElementById('alert-message');
        const successMessage = document.getElementById('success-message');
        const postOfficeNameEl = document.getElementById('post-office-name');
        const affiliatedPostOfficeEl = document.getElementById('affiliated-post-office');
        const buttonText = document.getElementById('button-text');
        const buttonLoader = document.getElementById('button-loader');
        
        async function saveDataToFirestore() {
            const postOfficeName = postOfficeNameEl.value.trim();
            const affiliatedPostOffice = affiliatedPostOfficeEl.value.trim();

            if (!window.currentLat || !window.currentLng || !postOfficeName || !affiliatedPostOffice) {
                successMessage.classList.add('hidden');
                alertMessage.textContent = "กรุณากรอกข้อมูลให้ครบทุกช่อง และปักหมุดบนแผนที่";
                alertMessage.classList.remove('hidden');
                return;
            }
            
            alertMessage.classList.add('hidden');
            submitBtn.disabled = true;
            buttonText.textContent = 'กำลังบันทึก...';
            buttonLoader.classList.remove('hidden');

            try {
                await addDoc(collection(db, "locations"), {
                    latitude: parseFloat(window.currentLat),
                    longitude: parseFloat(window.currentLng),
                    post_office_name: postOfficeName,
                    affiliated_post_office: affiliatedPostOffice,
                    timestamp: new Date()
                });

                successMessage.textContent = 'บันทึกข้อมูลเรียบร้อยแล้ว ขอบคุณครับ!';
                successMessage.classList.remove('hidden');
                
                postOfficeNameEl.value = '';
                affiliatedPostOfficeEl.value = '';
                
                setTimeout(() => {
                    successMessage.classList.add('hidden');
                }, 4000);

            } catch (e) {
                console.error("Error adding document: ", e);
                alertMessage.textContent = "เกิดข้อผิดพลาดในการบันทึกข้อมูล";
                alertMessage.classList.remove('hidden');
            } finally {
                submitBtn.disabled = false;
                buttonText.textContent = 'บันทึกข้อมูล';
                buttonLoader.classList.add('hidden');
            }
        }
        
        submitBtn.addEventListener('click', saveDataToFirestore);
    </script>

        function initMap() {
            // Changed fallback position to Ubon Ratchathani
            const fallbackPosition = { lat: 15.228, lng: 104.856 };

            const createMapAndMarker = (position, zoomLevel) => {
                const map = new google.maps.Map(document.getElementById("map"), {
                    zoom: zoomLevel,
                    center: position,
                    mapTypeControl: false,
                    streetViewControl: false,
                });

                const marker = new google.maps.Marker({
                    position: position,
                    map: map,
                    draggable: true,
                    title: "ลากหมุดนี้ไปยังตำแหน่งของคุณ"
                });
                
                window.currentLat = position.lat.toFixed(6);
                window.currentLng = position.lng.toFixed(6);
                latitudeEl.textContent = window.currentLat;
                longitudeEl.textContent = window.currentLng;

                marker.addListener("dragend", () => {
                    const newPosition = marker.getPosition();
                    window.currentLat = newPosition.lat().toFixed(6);
                    window.currentLng = newPosition.lng().toFixed(6);
                    latitudeEl.textContent = window.currentLat;
                    longitudeEl.textContent = window.currentLng;
                });
            };

            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(
                    (position) => {
                        createMapAndMarker({ lat: position.coords.latitude, lng: position.coords.longitude }, 16);
                    },
                    () => {
                        // Error getting location, likely due to iframe restrictions
                        mainContent.classList.add('blur-sm');
                        locationModal.classList.remove('hidden');
                        // Use Ubon as fallback and zoom in closer
                        createMapAndMarker(fallbackPosition, 10); 
                    }
                );
            } else {
                // Browser doesn't support Geolocation
                createMapAndMarker(fallbackPosition, 10);
            }
        }
